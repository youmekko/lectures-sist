--QABoard 테이블 설계
--항목 추출-> 번호, 제목, 내용, 비밀글, 작성일, 블라인드, ClientIP, 답변(자식글), 패스워드
--테이블 추출->QABoard 테이블(PK), 답변글 테이블(FK), 패스워드테이블(FK)

CREATE TABLE QABoardTable (
	qid VARCHAR(4) /* Q001 형태 */
	, title VARCHAR(100)
	, content VARCHAR(1000)
	, writeday DATE
	, clientip VARCHAR(15)
	, blind int /* 기본값 0, 블라인드 1 */
	, privacy int /* 기본값 0, 비밀글 1 */
);


CREATE TABLE QABoardReplyTable (
	qid VARCHAR(4)   /* PK and FK */
	, replyContent VARCHAR(1000)
);


CREATE TABLE QABoardPasswordTable (
	qid VARCHAR(4)   /* PK and FK */
	, pw VARCHAR(30)
);


ALTER TABLE QABoardTable
   ADD CONSTRAINT QABoardTable_qid_pk PRIMARY KEY(qid);
   
ALTER TABLE QABoardReplyTable
   ADD CONSTRAINT QABoardReplyTable_qid_pk  PRIMARY KEY(qid);
   
ALTER TABLE QABoardPasswordTable
   ADD CONSTRAINT QABoardPasswordTable_qid_pk  PRIMARY KEY(qid);  
   
ALTER TABLE QABoardReplyTable
   ADD CONSTRAINT QABoardReplyTable_qid_fk FOREIGN KEY (qid)
         REFERENCES QABoardTable (qid);
         
ALTER TABLE QABoardPasswordTable
   ADD CONSTRAINT QABoardPasswordTable_qid_fk FOREIGN KEY (qid)
         REFERENCES QABoardTable (qid);
         
INSERT INTO QABoardTable (qid, title, content, writeday, clientip, blind, privacy)
	VALUES ((SELECT * FROM (SELECT CONCAT('Q', LPAD(IFNULL(SUBSTRING(MAX(qid), 2), 0) + 1, 3, 0)) AS newQid FROM QABoardTable) q), '문의게시판이 오픈되었습니다.', '회원 가입 없이 자유롭게 글을 쓸 수 있습니다.\r\n 빠른 답변을 달 수 있도록 하겠습니다.\r\n - 관리자 - ', NOW(), '127.0.0.1', 0, 0);
    
SELECT qid, title, content, writeday, clientip, blind, privacy 
	FROM QABoardTable;

SELECT CONCAT('Q', LPAD(IFNULL(SUBSTRING(MAX(qid), 2), 0) + 1, 3, 0)) AS qid 
	FROM QABoardTable;


SELECT qid, title, content, writeday, clientip, blind, privacy
	
	, (SELECT replyContent FROM qaboardreplytable WHERE qid=qa.qid) AS replyContent
	FROM QABoardTable qa
	WHERE blind=0
	ORDER BY qid DESC;
